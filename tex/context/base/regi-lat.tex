%D \ASCII\ characters. Some more definitions will be added. 

\startregime[latin2]

\defineactivetoken æ {\cacute}
\defineactivetoken Æ {\Cacute} 

\defineactivetoken è {\ccaron} 
\defineactivetoken È {\Ccaron}
\defineactivetoken ¹ {\scaron}  
\defineactivetoken © {\Scaron} 
\defineactivetoken ¾ {\zcaron} 
\defineactivetoken ® {\Zcaron} 

\defineactivetoken ð {\pseudoencodeddj}
\defineactivetoken Ð {\pseudoencodedDJ}

\stopregime

\endinput
