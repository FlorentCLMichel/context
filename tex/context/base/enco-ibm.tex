%D This module activates the \kap{IBM PC} high \ASCII\ 
%D characters, such as â and ·. 

\startregime [ibm]

\defineactivecharacter â {\"e} 
\defineactivecharacter Ç {\'e}
\defineactivecharacter ä {\`e}
\defineactivecharacter à {\^e} 

\defineactivecharacter Ñ {\"a}
\defineactivecharacter † {\'a}
\defineactivecharacter Ö {\`a}
\defineactivecharacter É {\^a}

\defineactivecharacter î {\"o}
\defineactivecharacter ¢ {\'o}
\defineactivecharacter ï {\`o}
\defineactivecharacter ì {\^o}

\defineactivecharacter ã {\"\dotlessi}
\defineactivecharacter ° {\'\dotlessi}
\defineactivecharacter ç {\`\dotlessi}
\defineactivecharacter å {\^\dotlessi}

\defineactivecharacter Å {\"u}
\defineactivecharacter £ {\'u}
\defineactivecharacter ó {\`u}
\defineactivecharacter ñ {\^u}

\defineactivecharacter ê {\'E}
\defineactivecharacter é {\"A}
\defineactivecharacter ö {\"U}

\defineactivecharacter á {\c c}
\defineactivecharacter Ä {\c C}

\defineactivecharacter § {\~n}

\defineactivecharacter · {\SS}  

\defineactivecharacter Æ {\ifvmode\leavevmode\fi\leftguillemot\prewordbreak}
\defineactivecharacter Ø {\prewordbreak\rightguillemot}

\stopregime 

\enableregime[windows]
