\def\ilencodedrA%
  {\leavevmode\hbox\bgroup
   \setbox0\hbox{h}%
   \dimen0=\ht0
   \advance\dimen0 by -1ex
   \rlap{\raise.67\dimen0\hbox{\char'27}}A%
   \egroup}

\def\ilencodedl%
  {{\char32l}}

\def\ilencodedL% I don't like the \leavevmode here, better \enterhmode
  {\leavevmode\hbox\bgroup
   \setbox0\hbox{L}%
   \hbox to\wd0{\hss\char32L}%
   \egroup}
